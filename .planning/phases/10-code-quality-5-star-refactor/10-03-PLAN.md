# Plan 03: Frontend Structural Splits — Sidebar, MessageList, Sanitizers, IconButton

## Goal
Split the two largest frontend components (`chat-sidebar.tsx` at 329 LOC, `chat-message-list.tsx` at 276 LOC) into focused sub-components, create a shared `content-sanitizers.ts` utility, extract an `IconButton` reusable component, and remove the entirely dead `llm-response-formatter.ts`.

## must_haves
- `chat-sidebar.tsx` is < 150 LOC, composed from `ChatHistorySection`, `SearchSection`, and `useChatGrouping` hook
- `chat-message-list.tsx` is < 150 LOC, composed from `AttachmentRenderer`, `BotMessageContent`, and `ChatMessageContext`
- `content-sanitizers.ts` exists with `sanitizeBotContent()`, `stripCodeBlocks()`, `stripPythonTracebacks()`, `deduplicateCitations()`, `cleanSystemNotes()`
- `llm-response-formatter.ts` is deleted (0 imports anywhere in codebase)
- `icon-button.tsx` exists as reusable `Tooltip + Button` wrapper
- `ChatMessageContext.tsx` provides React Context replacing 9 prop-drilled values
- All extracted components are properly typed with exported interfaces

## threat_model
- **Component prop changes**: Extracting sub-components risks breaking parent-child interfaces. Mitigation: the parent component re-exports from new locations and passes the same props it always did.
- **React Context performance**: Adding `ChatMessageContext` could cause unnecessary re-renders if the value object changes identity on every render. Mitigation: memoize the context value with `useMemo`.
- **Dead code removal**: `llm-response-formatter.ts` was confirmed as 0 imports across the codebase. Safe to delete. `cleanSystemNotes()` logic is duplicated in `chat-message-list.tsx:sanitizeBotContent()` so no functionality is lost.

```typescript
/**
 * Content sanitization pipeline for LLM bot responses.
 *
 * Consolidates sanitization logic previously duplicated in
 * chat-message-list.tsx and llm-response-formatter.ts.
 */

/** Remove SQL/Python code blocks that LLM might accidentally include. */
export function stripCodeBlocks(content: string): string {
  // Copy the exact regex from chat-message-list.tsx sanitizeBotContent
  // that removes ```sql...``` and ```python...``` blocks
  return content
    .replace(/```(?:sql|SQL)[\s\S]*?```/g, '')
    .replace(/```(?:python|Python)[\s\S]*?```/g, '');
}

/** Remove Python tracebacks (Traceback, File "...", Error: ...). */
export function stripPythonTracebacks(content: string): string {
  return content.replace(/Traceback \(most recent call last\)[\s\S]*?(?:Error|Exception):.*$/gm, '');
}

/** Deduplicate citations that reference the same filename+page. */
export function deduplicateCitations(content: string): string {
  // Copy the exact deduplication logic from chat-message-list.tsx
  // (the regex-based approach that finds [source: filename, page X])
  const seen = new Set<string>();
  return content.replace(
    /\[(?:nguồn|source|Nguồn|Source):\s*([^\]]+)\]/g,
    (match, citation) => {
      const key = citation.trim().toLowerCase();
      if (seen.has(key)) return '';
      seen.add(key);
      return match;
    },
  );
}

/** Remove unwanted system notes (Vietnamese advisory text). */
export function cleanSystemNotes(content: string): string {
  return content
    .replace(/Lưu ý: Đây là thông tin thô từ tài liệu[^\n]*/g, '')
    .replace(/Lưu ý:.*bạn có thể hỏi cụ thể hơn[^\n]*/g, '');
}

/** Remove malformed markdown table artifacts from legacy messages. */
export function stripMalformedTables(content: string): string {
  // Copy the exact regex from chat-message-list.tsx that strips orphan pipes
  return content.replace(/^\|[-\s|]*\|$/gm, '');
}

/**
 * Full sanitization pipeline for bot responses.
 * Applies all sanitizers in sequence.
 */
export function sanitizeBotContent(content: string): string {
  let text = content;
  text = stripCodeBlocks(text);
  text = stripPythonTracebacks(text);
  text = deduplicateCitations(text);
  text = cleanSystemNotes(text);
  text = stripMalformedTables(text);
  return text.trim();
}
```

```typescript
import { useMemo } from 'react';
import { MS_PER_DAY } from '@/lib/constants';

/** Date group labels for sidebar. */
function formatDateLabel(date: Date, now: Date): string {
  const diff = now.getTime() - date.getTime();
  if (diff < MS_PER_DAY) return 'Hôm nay';
  if (diff < 2 * MS_PER_DAY) return 'Hôm qua';
  return 'Khác';
}

export interface ChatGroupItem {
  id: string;
  title: string | null;
  updated_at: string;
}

export interface ChatGroup {
  label: string;
  items: ChatGroupItem[];
}

/**
 * Groups chat sessions by date (Hôm nay / Hôm qua / Khác).
 *
 * Pure computation extracted from ChatSidebar.
 */
export function useChatGrouping(chats: ChatGroupItem[]): ChatGroup[] {
  return useMemo(() => {
    const now = new Date();
    const groups = new Map<string, ChatGroupItem[]>();

    for (const chat of chats) {
      const label = formatDateLabel(new Date(chat.updated_at), now);
      const group = groups.get(label) ?? [];
      group.push(chat);
      groups.set(label, group);
    }

    // Copy the exact ordering logic from the original — preserve order: today, yesterday, other
    const order = ['Hôm nay', 'Hôm qua', 'Khác'];
    return order
      .filter((label) => groups.has(label))
      .map((label) => ({ label, items: groups.get(label)! }));
  }, [chats]);
}
```

```typescript
'use client';

import { createContext, useContext, useMemo, type ReactNode } from 'react';
import type { Attachment, BranchInfo } from '@/services/chat-backend';

export interface ChatMessageContextValue {
  onPreviewAttachment?: (att: Attachment) => void;
  branchInfoMap: Record<string, BranchInfo>;
  onNavigateBranch?: (messageId: string, direction: 'prev' | 'next') => void;
  onEditMessage?: (messageId: string, newContent: string) => void;
  onRegenerateMessage?: (messageId: string) => void;
  editingMessageId: string | null;
  onStartEdit?: (messageId: string) => void;
  onCancelEdit?: () => void;
  branchLoading?: boolean;
}

const ChatMessageCtx = createContext<ChatMessageContextValue | null>(null);

export function useChatMessageContext(): ChatMessageContextValue {
  const ctx = useContext(ChatMessageCtx);
  if (!ctx) {
    throw new Error('useChatMessageContext must be used within ChatMessageProvider');
  }
  return ctx;
}

export function ChatMessageProvider({
  children,
  value,
}: {
  children: ReactNode;
  value: ChatMessageContextValue;
}) {
  const memoized = useMemo(() => value, [
    value.branchInfoMap,
    value.editingMessageId,
    value.branchLoading,
    value.onPreviewAttachment,
    value.onNavigateBranch,
    value.onEditMessage,
    value.onRegenerateMessage,
    value.onStartEdit,
    value.onCancelEdit,
  ]);
  return (
    <ChatMessageCtx.Provider value={memoized}>
      {children}
    </ChatMessageCtx.Provider>
  );
}
```

```typescript
import type { LucideIcon } from 'lucide-react';
import { Button, type ButtonProps } from '@/components/ui/button';
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from '@/components/ui/tooltip';

export interface IconButtonProps extends Omit<ButtonProps, 'children'> {
  icon: LucideIcon;
  tooltip: string;
  iconClassName?: string;
}

/**
 * Button with icon and tooltip — replaces repeated
 * TooltipProvider > Tooltip > TooltipTrigger > Button > Icon pattern.
 */
export function IconButton({
  icon: Icon,
  tooltip,
  iconClassName,
  variant = 'ghost',
  size = 'icon',
  className = 'h-7 w-7',
  ...props
}: IconButtonProps) {
  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>
          <Button variant={variant} size={size} className={className} {...props}>
            <Icon className={iconClassName} />
            <span className="sr-only">{tooltip}</span>
          </Button>
        </TooltipTrigger>
        <TooltipContent>
          <p>{tooltip}</p>
        </TooltipContent>
      </Tooltip>
    </TooltipProvider>
  );
}
```
