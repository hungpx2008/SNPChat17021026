```python
@router.post("/{session_id}/messages/{message_id}/edit", response_model=BranchMessageSchema)
async def edit_message(
    session_id: UUID,
    message_id: UUID,
    payload: EditMessageRequest,
    db_session: ChatSession = Depends(get_session_or_404),
    db: AsyncSession = Depends(get_db_session),
) -> Any:
    """Fork conversation by editing a user message with new content.
    
    Creates a new sibling user message at the same parent. The frontend
    is responsible for triggering the AI response (same as normal message flow).
    We also dispatch process_chat_response to chunk+embed the new user message.
    """
    tree = ConversationTree(db)
    try:
        new_message = await tree.edit_message(message_id, payload.content)
    except ValueError as e:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail=str(e))
    await db.commit()

    # Chunk+embed the new user message (this is what process_chat_response does)
    from src.worker.tasks import process_chat_response
    process_chat_response.delay(
        session_id=str(session_id),
        message_id=str(new_message.id),
        content=payload.content,
        role="user",
        user_id=db_session.user_id,
        department=db_session.department,
    )

    service = ChatService(db)
    result = service.serialize_message(new_message)
    # Signal frontend that it needs to trigger AI response (same as normal flow)
    result["needs_ai_response"] = True
    return result


@router.post(
    "/{session_id}/messages/{message_id}/regenerate",
    response_model=BranchMessageSchema,
)
async def regenerate_message(
    session_id: UUID,
    message_id: UUID,
    db_session: ChatSession = Depends(get_session_or_404),
    db: AsyncSession = Depends(get_db_session),
) -> Any:
    """Fork conversation by regenerating an AI response.
    
    Creates a placeholder AI message. The Celery task fills the placeholder
    content. Supports rag and sql modes by reading the original AI message's
    metadata to determine which task to dispatch.
    """
    tree = ConversationTree(db)
    try:
        new_ai_message = await tree.regenerate(message_id)
    except ValueError as e:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST, detail=str(e)
        )
    await db.commit()

    # Find the parent user message to get its content for regeneration
    from src.repositories.messages import MessageRepository
    msg_repo = MessageRepository(db)
    parent_msg = None
    if new_ai_message.parent_message_id:
        parent_msg = await msg_repo.get_message_by_id(
            new_ai_message.parent_message_id
        )

    if parent_msg:
        # Determine which mode to use for regeneration.
        # Check original AI message metadata for clues about mode (rag/sql).
        original_ai = await msg_repo.get_message_by_id(message_id)
        original_meta = (original_ai.meta or {}) if original_ai else {}
        
        # Detect mode from original message metadata or content patterns
        mode = "chat"  # default
        if original_meta.get("rag_chunk_ids"):
            mode = "rag"
        elif original_meta.get("attachments") and any(
            a.get("type") == "chart" for a in original_meta.get("attachments", [])
        ):
            mode = "sql"

        if mode == "rag":
            from src.worker.tasks import rag_document_search
            rag_document_search.delay(
                question=parent_msg.content,
                session_id=str(session_id),
                user_id=db_session.user_id,
                department=db_session.department,
                target_message_id=str(new_ai_message.id),
            )
        elif mode == "sql":
            from src.worker.tasks import run_sql_query
            run_sql_query.delay(
                question=parent_msg.content,
                session_id=str(session_id),
                user_id=db_session.user_id,
                target_message_id=str(new_ai_message.id),
            )
        else:
            # Chat mode: frontend handles AI response via getHelp() + SSE
            # Signal frontend to trigger AI response
            pass

    service = ChatService(db)
    result = service.serialize_message(new_ai_message)
    result["needs_ai_response"] = (mode == "chat") if parent_msg else False
    result["mode"] = mode if parent_msg else "chat"
    return result


@router.get(

async def get_branch_info(

        )
    return {
        "current_index": info.current_index,
        "total_branches": info.total_branches,
        "sibling_ids": info.sibling_ids,
        "fork_point_id": info.fork_point_id,
    }


@router.post(
    "/{session_id}/messages/{message_id}/navigate",
    response_model=list[BranchMessageSchema],
)
async def navigate_branch(
    session_id: UUID,
    message_id: UUID,
    payload: NavigateBranchRequest,
    db_session: ChatSession = Depends(get_session_or_404),
    db: AsyncSession = Depends(get_db_session),
) -> Any:
    """Switch to adjacent branch (direction: -1 or +1)."""
    tree = ConversationTree(db)
    try:
        active_messages = await tree.navigate_branch(

    except ValueError as e:

        )
    await db.commit()

    service = ChatService(db)
    return [service.serialize_message(msg) for msg in active_messages]


@router.get("/{session_id}/tree", response_model=ConversationTreeSchema)
```
