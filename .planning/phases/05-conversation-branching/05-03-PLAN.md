<action>
Create new file `chatSNP170226/frontend/src/hooks/use-conversation-tree.ts`:

**W11 FIX:** `fetchAllBranchInfo` makes N parallel API calls. To mitigate performance for large conversations, we: (1) only fetch branch info for messages that have a `parentMessageId` or are root messages (skip messages without backendId), (2) use `Promise.allSettled` with a concurrency limiter, and (3) add a comment acknowledging the trade-off. A batch endpoint could be added in a future iteration for better performance, but for now N calls is acceptable since conversations rarely exceed 50 messages on the active branch, and the endpoint is lightweight (single DB query).

```typescript
import { useCallback, useState } from "react";
import { chatBackend, type BranchInfo, type BackendMessage } from "@/services/chat-backend";
import type { Message } from "@/components/chat/types";

interface BranchState {
  /** Map from backendId -> BranchInfo */
  branchInfoMap: Record<string, BranchInfo>;
  /** Currently editing message backendId, or null */
  editingMessageId: string | null;
  /** Loading state for branch operations */
  loading: boolean;
}

export function useConversationTree(
  sessionId: string | null,
  setMessages: React.Dispatch<React.SetStateAction<Message[]>>,
  mapBackendMessage: (msg: BackendMessage) => Message,
) {
  const [branchState, setBranchState] = useState<BranchState>({
    branchInfoMap: {},
    editingMessageId: null,
    loading: false,
  });

  const fetchBranchInfo = useCallback(
    async (messageId: string) => {
      if (!sessionId) return null;
      try {
        const info = await chatBackend.getBranchInfo(sessionId, messageId);
        setBranchState((prev) => ({
          ...prev,
          branchInfoMap: { ...prev.branchInfoMap, [messageId]: info },
        }));
        return info;
      } catch (err) {
        console.error("Failed to fetch branch info", err);
        return null;
      }
    },
    [sessionId],
  );

  const fetchAllBranchInfo = useCallback(
    async (messages: Message[]) => {
      if (!sessionId) return;
      // W11: Only fetch for messages with backendIds. Active branch conversations
      // typically have <50 messages. Each getBranchInfo call is a lightweight DB query.
      // For very large conversations, a batch endpoint could be added later.
      const backendIds = messages
        .map((m) => m.backendId)
        .filter((id): id is string => !!id);

      const infoMap: Record<string, BranchInfo> = {};
      await Promise.allSettled(
        backendIds.map(async (id) => {
          try {
            const info = await chatBackend.getBranchInfo(sessionId, id);
            if (info.total_branches > 1) {
              infoMap[id] = info;
            }
          } catch {
            // ignore — message may not have branches
          }
        }),
      );

      setBranchState((prev) => ({
        ...prev,
        branchInfoMap: { ...prev.branchInfoMap, ...infoMap },
      }));
    },
    [sessionId],
  );

  const navigateBranch = useCallback(
    async (messageId: string, direction: -1 | 1) => {
      if (!sessionId) return;
      setBranchState((prev) => ({ ...prev, loading: true }));
      try {
        const newMessages = await chatBackend.navigateBranch(
          sessionId,
          messageId,
          direction,
        );
        const mapped = newMessages.map(mapBackendMessage);
        setMessages(mapped.length > 0 ? mapped : []);

        // Refresh branch info for all messages after navigation
        await fetchAllBranchInfo(mapped);
      } catch (err) {
        console.error("Failed to navigate branch", err);
      } finally {
        setBranchState((prev) => ({ ...prev, loading: false }));
      }
    },
    [sessionId, mapBackendMessage, setMessages, fetchAllBranchInfo],
  );

  const editMessage = useCallback(
    async (messageId: string, newContent: string) => {
      if (!sessionId) return;
      setBranchState((prev) => ({
        ...prev,
        loading: true,
        editingMessageId: null,
      }));
      try {
        await chatBackend.editMessage(sessionId, messageId, newContent);
        // After edit, reload session messages (active branch will be updated)
        // The caller (chat-ui) should handle SSE for the AI response
        return true;
      } catch (err) {
        console.error("Failed to edit message", err);
        return false;
      } finally {
        setBranchState((prev) => ({ ...prev, loading: false }));
      }
    },
    [sessionId],
  );

  const regenerateMessage = useCallback(
    async (messageId: string) => {
      if (!sessionId) return;
      setBranchState((prev) => ({ ...prev, loading: true }));
      try {
        await chatBackend.regenerateMessage(sessionId, messageId);
        // The caller (chat-ui) should handle SSE for the new AI response
        return true;
      } catch (err) {
        console.error("Failed to regenerate message", err);
        return false;
      } finally {
        setBranchState((prev) => ({ ...prev, loading: false }));
      }
    },
    [sessionId],
  );

  const startEditing = useCallback((messageId: string) => {
    setBranchState((prev) => ({ ...prev, editingMessageId: messageId }));
  }, []);

  const cancelEditing = useCallback(() => {
    setBranchState((prev) => ({ ...prev, editingMessageId: null }));
  }, []);

  return {
    branchInfoMap: branchState.branchInfoMap,
    editingMessageId: branchState.editingMessageId,
    branchLoading: branchState.loading,
    fetchBranchInfo,
    fetchAllBranchInfo,
    navigateBranch,
    editMessage,
    regenerateMessage,
    startEditing,
    cancelEditing,
  };
}
```

```tsx
'use client';

import { ChevronLeft, ChevronRight } from "lucide-react";
import { Button } from "@/components/ui/button";
import type { BranchInfo } from "@/services/chat-backend";

interface BranchNavigatorProps {
  branchInfo: BranchInfo;
  onNavigate: (direction: -1 | 1) => void;
  disabled?: boolean;
}

export function BranchNavigator({
  branchInfo,
  onNavigate,
  disabled = false,
}: BranchNavigatorProps) {
  if (branchInfo.total_branches <= 1) return null;

  return (
    <div className="inline-flex items-center gap-0.5 text-xs text-muted-foreground">
      <Button
        variant="ghost"
        size="icon"
        className="h-5 w-5"
        onClick={() => onNavigate(-1)}
        disabled={disabled || branchInfo.current_index <= 1}
        aria-label="Nhanh truoc"
      >
        <ChevronLeft size={12} />
      </Button>
      <span className="min-w-[3ch] text-center tabular-nums">
        {branchInfo.current_index}/{branchInfo.total_branches}
      </span>
      <Button
        variant="ghost"
        size="icon"
        className="h-5 w-5"
        onClick={() => onNavigate(1)}
        disabled={disabled || branchInfo.current_index >= branchInfo.total_branches}
        aria-label="Nhanh sau"
      >
        <ChevronRight size={12} />
      </Button>
    </div>
  );
}
```

```tsx
import { useState } from "react";
import { Pencil, RefreshCw, Check, X } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";

interface MessageActionsProps {
  role: "user" | "bot";
  backendId: string;
  content: string;
  onEdit: (messageId: string, newContent: string) => void;
  onRegenerate: (messageId: string) => void;
  editing: boolean;
  onStartEdit: (messageId: string) => void;

  backendId,
  content,
  onEdit,
  onRegenerate,
  editing,
  onStartEdit,
  onCancelEdit,
  disabled = false,

  if (editing) {
```
