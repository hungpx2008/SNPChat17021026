```python
async def get_last_active_message(self, session_id: UUID) -> ChatMessage | None:
    """Get the most recent is_active_branch=True message in a session.
    
    Used for parent_message_id assignment on new messages. After branching,
    only messages on the active branch should be considered as potential parents.
    Falls back to any last message if no active messages exist (pre-migration data).
    """
    stmt = (
        select(ChatMessage)
        .where(
            ChatMessage.session_id == session_id,
            ChatMessage.is_active_branch.is_(True),
        )
        .order_by(ChatMessage.created_at.desc())
        .limit(1)
    )
    result = await self.session.execute(stmt)
    msg = result.scalar_one_or_none()
    if msg is not None:
        return msg
    # Fallback for pre-migration data without is_active_branch
    stmt_fallback = (
        select(ChatMessage)
        .where(ChatMessage.session_id == session_id)
        .order_by(ChatMessage.created_at.desc())
        .limit(1)
    )
    result = await self.session.execute(stmt_fallback)
    return result.scalar_one_or_none()
```

```python
async def list_active_branch_messages(
    self, session_id: UUID
) -> list[ChatMessage]:
    """Walk the conversation tree following is_active_branch=True."""
    from sqlalchemy import text as sa_text

    # Use recursive CTE to walk the active branch
    cte_sql = sa_text("""
        WITH RECURSIVE active_branch AS (
            -- Base: root message(s) with no parent in this session
            SELECT id, parent_message_id, session_id, role, content,
                   metadata, created_at, branch_index, is_active_branch
            FROM chat_messages
            WHERE session_id = :session_id
              AND parent_message_id IS NULL
              AND is_active_branch = TRUE

            UNION ALL

            -- Recurse: follow active children
            SELECT cm.id, cm.parent_message_id, cm.session_id, cm.role, cm.content,
                   cm.metadata, cm.created_at, cm.branch_index, cm.is_active_branch
            FROM chat_messages cm
            INNER JOIN active_branch ab ON cm.parent_message_id = ab.id
            WHERE cm.is_active_branch = TRUE
        )
        SELECT id FROM active_branch ORDER BY created_at ASC
    """)
    result = await self.session.execute(cte_sql, {"session_id": session_id})
    active_ids = [row[0] for row in result.fetchall()]

    if not active_ids:
        # Fallback: return all messages (no tree structure yet)
        return await self.list_messages(session_id)

    # Fetch full ORM objects in order
    stmt = (
        select(ChatMessage)
        .where(ChatMessage.id.in_(active_ids))
        .order_by(ChatMessage.created_at.asc())
    )
    result = await self.session.execute(stmt)
    return list(result.scalars())
```

```python
from __future__ import annotations

import json
import logging
from dataclasses import dataclass, field
from uuid import UUID

from sqlalchemy.ext.asyncio import AsyncSession

from src.core.redis_client import get_redis
from src.models.models import ChatMessage
from src.repositories.messages import MessageRepository

logger = logging.getLogger(__name__)


@dataclass
class BranchInfo:
    """Branch navigation info for a single message node."""
    current_index: int  # 1-based
    total_branches: int
    sibling_ids: list[str] = field(default_factory=list)
    fork_point_id: str | None = None


class ConversationTree:
    """Service for conversation branching: edit, regenerate, navigate."""

    def __init__(self, session: AsyncSession):
        self.session = session
        self.message_repo = MessageRepository(session)
        self.redis = get_redis()

    async def get_active_branch(self, session_id: UUID) -> list[ChatMessage]:
        """Get messages along the active branch using recursive CTE."""
        return await self.message_repo.list_active_branch_messages(session_id)

    async def get_branch_info(self, message_id: UUID) -> BranchInfo | None:
        """Get sibling count and navigation info for a message."""
        message = await self.message_repo.get_message_by_id(message_id)
        if message is None:
            return None

        if message.parent_message_id is None:
            # Root message — get root siblings in same session
            roots = await self.message_repo.get_root_messages(message.session_id)
            sibling_ids = [str(r.id) for r in roots]
            current_idx = next(
                (i + 1 for i, r in enumerate(roots) if r.id == message.id), 1
            )
            return BranchInfo(
                current_index=current_idx,
                total_branches=len(roots),
                sibling_ids=sibling_ids,
                fork_point_id=None,
            )

        siblings = await self.message_repo.get_siblings(message.parent_message_id)
        sibling_ids = [str(s.id) for s in siblings]
        current_idx = next(
            (i + 1 for i, s in enumerate(siblings) if s.id == message.id), 1
        )
        return BranchInfo(
            current_index=current_idx,
            total_branches=len(siblings),
            sibling_ids=sibling_ids,
            fork_point_id=str(message.parent_message_id),
        )

    async def edit_message(
        self,
        message_id: UUID,
        new_content: str,
    ) -> ChatMessage:
        """Fork: create new sibling at same parent with new content.

        1. Deactivate old message + its descendants
        2. Create new message at same parent with next branch_index
        3. Return new message (caller should trigger AI response)
        """
        original = await self.message_repo.get_message_by_id(message_id)
        if original is None:
            raise ValueError(f"Message {message_id} not found")

        # Deactivate old message and its entire subtree
        await self._deactivate_subtree(message_id)

        # Calculate next branch_index among siblings
        if original.parent_message_id:
            siblings = await self.message_repo.get_siblings(
                original.parent_message_id
            )
        else:
            siblings = await self.message_repo.get_root_messages(
                original.session_id
            )
        next_index = max(s.branch_index for s in siblings) + 1 if siblings else 0

        # Create new message at same parent
        new_message = await self.message_repo.create_message(
            session_id=original.session_id,
            role=original.role,
            content=new_content,
            metadata=original.meta or {},
            parent_message_id=original.parent_message_id,
            branch_index=next_index,
            is_active_branch=True,
        )

        await self.session.flush()

        # Invalidate Redis cache
        await self._invalidate_cache(original.session_id)

        return new_message

    async def regenerate(self, ai_message_id: UUID) -> ChatMessage:
        """Fork: create new AI response placeholder at same parent.

        1. Find the AI message's parent (the user message)
        2. Deactivate old AI message subtree
        3. Create new AI message placeholder at same parent
        4. Return new AI message (caller should trigger AI task to fill content)
        """
        ai_message = await self.message_repo.get_message_by_id(ai_message_id)
        if ai_message is None:
            raise ValueError(f"Message {ai_message_id} not found")

        if ai_message.role != "assistant":
            raise ValueError("Chỉ có thể tạo lại phản hồi của trợ lý")

        # Deactivate old AI message subtree
        await self._deactivate_subtree(ai_message_id)

        # Get siblings to calculate next branch_index
        parent_id = ai_message.parent_message_id
        if parent_id:
            siblings = await self.message_repo.get_siblings(parent_id)
        else:
            siblings = await self.message_repo.get_root_messages(
                ai_message.session_id
            )
        next_index = max(s.branch_index for s in siblings) + 1 if siblings else 0

        # Create placeholder AI message — Celery task will update content via
        # update_message_content() or API call with target_message_id
        new_ai_message = await self.message_repo.create_message(
            session_id=ai_message.session_id,
            role="assistant",
            content="",  # placeholder — Celery task will update
            metadata={"regenerated_from": str(ai_message_id)},
            parent_message_id=parent_id,
            branch_index=next_index,
            is_active_branch=True,
        )

        await self.session.flush()
        await self._invalidate_cache(ai_message.session_id)

        return new_ai_message

    async def navigate_branch(
        self,
        message_id: UUID,
        direction: int,
    ) -> list[ChatMessage]:
        """Switch active branch. direction: -1 (prev) or +1 (next).

        1. Get siblings of this message
        2. Find target sibling (current_index + direction)
        3. Deactivate current message subtree, activate target subtree
        4. Return new active branch messages for the session
        """
        info = await self.get_branch_info(message_id)
        if info is None:
            raise ValueError(f"Message {message_id} not found")

        target_index = info.current_index - 1 + direction  # convert to 0-based
        if target_index < 0 or target_index >= info.total_branches:
            raise ValueError("Không có nhánh nào ở hướng này")

        target_id = UUID(info.sibling_ids[target_index])
        current_msg = await self.message_repo.get_message_by_id(message_id)
        if current_msg is None:
            raise ValueError(f"Message {message_id} not found")

        # Deactivate current subtree (from message_id downward)
        await self._deactivate_subtree(message_id)

        # Activate target subtree
        await self._activate_subtree(target_id)

        await self.session.flush()
        await self._invalidate_cache(current_msg.session_id)

        return await self.get_active_branch(current_msg.session_id)

    async def get_full_tree(self, session_id: UUID) -> dict:
        """Get full conversation tree structure for visualization."""
        all_messages = await self.message_repo.list_messages(session_id)

        nodes: dict[str, dict] = {}
        for msg in all_messages:
            nodes[str(msg.id)] = {
                "id": str(msg.id),
                "role": msg.role,
                "content": msg.content[:100],  # truncate for tree view
                "parent_id": str(msg.parent_message_id)
                if msg.parent_message_id
                else None,
                "branch_index": msg.branch_index,
                "is_active": msg.is_active_branch,
                "created_at": msg.created_at.isoformat(),
                "children": [],
            }

        # Build parent-child relationships
        roots: list[dict] = []
        for node in nodes.values():
            parent_id = node["parent_id"]
            if parent_id and parent_id in nodes:
                nodes[parent_id]["children"].append(node)
            elif parent_id is None:
                roots.append(node)

        return {"session_id": str(session_id), "roots": roots}

    # ── Internal helpers ──────────────────────────────────────────

    async def _deactivate_subtree(self, message_id: UUID) -> None:
        """Set is_active_branch=False for a message and all its descendants."""
        descendant_ids = await self._collect_subtree_ids(message_id)
        await self.message_repo.update_active_branch(descendant_ids, is_active=False)

    async def _activate_subtree(self, message_id: UUID) -> None:
        """Set is_active_branch=True for a message and its active descendants.

        Only activates ONE path down (the first/active child at each level).
        """
        # Activate the target message itself
        await self.message_repo.update_active_branch([message_id], is_active=True)

        # Walk down: at each level, activate the child with lowest branch_index
        current_id = message_id
        while True:
            children = await self.message_repo.get_siblings(current_id)
            if not children:
                break
            # Pick child with lowest branch_index (default path)
            target_child = min(children, key=lambda c: c.branch_index)
            await self.message_repo.update_active_branch(
                [target_child.id], is_active=True
            )
            current_id = target_child.id

    async def _collect_subtree_ids(self, message_id: UUID) -> list[UUID]:
        """Collect message_id + all descendant IDs (BFS)."""
        result = [message_id]
        queue = [message_id]
        while queue:
            parent_id = queue.pop(0)
            children = await self.message_repo.get_siblings(parent_id)
            for child in children:
                result.append(child.id)
                queue.append(child.id)
        return result

    async def _invalidate_cache(self, session_id: UUID) -> None:
        """Delete Redis cache for a session after branch operations."""
        cache_key = f"chat:session:{session_id}"
        try:
```
